PHOTOGRAPHER_EMAIL: 'jaydenbottaro2305@gmail.com', // Email address bookings get CC'd to (optional, leave '' to disable) NOTIFY_EMAIL: '', }; // ---------------------- ENTRY POINTS ---------------------- function doGet(e) { const action = e.parameter.action; // Approve/decline links (clicked from your email) return a simple HTML page if (action === 'approve' || action === 'decline') { try { const html = action === 'approve' ? } else { intervals.push([ev.getStartTime(), ev.getEndTime()]); } }); }); return intervals; } function getAvailableSlots(dateStr, durationMinutes) { if (!durationMinutes) durationMinutes = 60; if (CONFIG.BLOCKED_DATES.indexOf(dateStr) > -1) return []; const date = new Date(dateStr + 'T00:00:00'); const weekday = date.getDay(); const hours = CONFIG.BUSINESS_HOURS[weekday]; if (!hours) return []; const now = new Date(); const minStart = new Date(now.getTime() + CONFIG.MIN_NOTICE_HOURS * 3600000); const maxDate = new Date(now.getTime() + CONFIG.MAX_ADVANCE_DAYS * 86400000); if (date > maxDate) return []; const [openH, openM] = hours[0].split(':').map(Number); const [closeH, closeM] = hours[1].split(':').map(Number); const dayOpen = new Date(dateStr + 'T00:00:00'); dayOpen.setHours(openH, openM, 0, 0); const dayClose = new Date(dateStr + 'T00:00:00'); dayClose.setHours(closeH, closeM, 0, 0); const busy = getBusyIntervals(dateStr).map(([s, e]) => [ new Date(s.getTime() - CONFIG.BUFFER_MINUTES * 60000), new Date(e.getTime() + CONFIG.BUFFER_MINUTES * 60000), ]); const slots = []; const stepMinutes = 30; // slot granularity let cursor = new Date(dayOpen); while (cursor.getTime() + durationMinutes * 60000 <= dayClose.getTime()) { const slotEnd = new Date(cursor.getTime() + durationMinutes * 60000); const overlapsBusy = busy.some(([bS, bE]) => cursor < bE && slotEnd > bS); const tooSoon = cursor < minStart; if (!overlapsBusy && !tooSoon) { slots.push(Utilities.formatDate(cursor, Session.getScriptTimeZone(), 'HH:mm')); } cursor = new Date(cursor.getTime() + stepMinutes * 60000); } return slots; } function getMonthAvailability(monthStr, durationMinutes) { // monthStr = 'YYYY-MM' const [y, m] = monthStr.split('-').map(Number); const daysInMonth = new Date(y, m, 0).getDate(); const result = {}; for (let d = 1; d <= daysInMonth; d++) { const dateStr = monthStr + '-' + String(d).padStart(2, '0'); const slots = getAvailableSlots(dateStr, durationMinutes || 60); result[dateStr] = slots.length; } return result; } // ---------------------- BOOKING ---------------------- function createBooking(data) { const required = ['packageId', 'date', 'time', 'name', 'email']; const end = new Date(start.getTime() + pkg.durationMinutes * 60000); // Re-check availability right before booking to prevent double-booking const freshSlots = getAvailableSlots(data.date, pkg.durationMinutes); if (freshSlots.indexOf(data.time) === -1) { return { ok: false, error: 'That slot was just taken. Please pick another time.' }; } const cal = CONFIG.BOOKING_CALENDAR_ID === 'primary' ? CalendarApp.getDefaultCalendar() : CalendarApp.getCalendarById(CONFIG.BOOKING_CALENDAR_ID); // Reserve the slot immediately as a PENDING event (no guest yet — client // isn't invited until you approve). This blocks the slot from other // clients right away while you decide. const title = 'PENDING — ' + pkg.name + ' — ' + data.name; const description = [ 'Status: PENDING — awaiting your approval', 'Package: ' + pkg.name, 'Client: ' + data.name, 'Email: ' + data.email, 'Phone: ' + (data.phone || '—'), 'Notes: ' + (data.notes || '—'), ].join('\n'); const event = cal.createEvent(title, start, end, { description: description }); try { event.setColor(CalendarApp.EventColor.YELLOW); } catch (e) {} // Generate a one-time token that authorizes approving/declining this // specific booking, and store the booking details against it. const token = Utilities.getUuid(); const record = { eventId: event.getId(), packageName: pkg.name, date: data.date, time: data.time, name: data.name, email: data.email, phone: data.phone || '', notes: data.notes || '', }; PropertiesService.getScriptProperties().setProperty('booking_' + token, JSON.stringify(record)); sendPhotographerRequestEmail(token, record); sendClientRequestReceivedEmail(record); return { ok: true, message: 'Request sent!', package: pkg.name, date: data.date, time: data.time, }; } function getBookingRecord(token) { const raw = PropertiesService.getScriptProperties().getProperty('booking_' + token); if (!raw) throw new Error('This link has already been used or has expired.'); return JSON.parse(raw); } function formatDisplayDate(dateStr, timeStr) { const d = new Date(dateStr + 'T' + timeStr + ':00'); return Utilities.formatDate(d, Session.getScriptTimeZone(), "EEEE, MMMM d 'at' h:mm a"); } function sendPhotographerRequestEmail(token, r) { const base = ScriptApp.getService().getUrl(); const approveUrl = base + '?action=approve&token=' + token; const declineUrl = base + '?action=decline&token=' + token; const when = formatDisplayDate(r.date, r.time); const body = 'New booking request:\n\n' + 'Package: ' + r.packageName + '\n' + 'When: ' + when + '\n' + 'Client: ' + r.name + '\n' + 'Email: ' + r.email + '\n' + 'Phone: ' + (r.phone || '—') + '\n' + 'Notes: ' + (r.notes || '—') + '\n\n' + 'Approve: ' + approveUrl + '\n\n' + 'Decline: ' + declineUrl + '\n\n' + "The time slot is already held on your calendar as PENDING so it won't double-book while you decide.";
